199void App::onGraphics3D(RenderDevice* rd, Array<shared_ptr<Surface> >& allSurfaces) {
200 if (! scene()) {
201 if ((submitToDisplayMode() == SubmitToDisplayMode::MAXIMIZE_THROUGHPUT) && (!rd->swapBuffersAutomatically())) {
202 swapBuffers();
203 }
204 rd->clear();
205 rd->pushState(); {
206 rd->setProjectionAndCameraMatrix(activeCamera()->projection(), activeCamera()->frame());
207 drawDebugShapes();
208 } rd->popState();
209 return;
210 }
211
212 BEGIN_PROFILER_EVENT("App::onGraphics3D");
213 GBuffer::Specification gbufferSpec = m_gbufferSpecification;
214
215 extendGBufferSpecification(gbufferSpec);
216 m_gbuffer->setSpecification(gbufferSpec);
217 m_gbuffer->resize(m_framebuffer->width(), m_framebuffer->height());
218 m_gbuffer->prepare(rd, activeCamera(), 0, -(float)previousSimTimeStep(), m_settings.hdrFramebuffer.depthGuardBandThickness, m_settings.hdrFramebuffer.colorGuardBandThickness);
219 debugAssertGLOk();
220
221 m_renderer->render(rd,
222 activeCamera(),
223 m_framebuffer,
224 scene()->lightingEnvironment().ambientOcclusionSettings.enabled ? m_depthPeelFramebuffer : nullptr,
225 scene()->lightingEnvironment(), m_gbuffer,
226 allSurfaces,
227 [&]() -> decltype(auto) { return scene()->tritree(); }); // decltype(auto) for correct return type deduction in the lambda.
228
229 // Debug visualizations and post-process effects
230 rd->pushState(m_framebuffer); {
231 // Call to make the App show the output of debugDraw(...)
232 rd->setProjectionAndCameraMatrix(activeCamera()->projection(), activeCamera()->frame());
233 drawDebugShapes();
234 const shared_ptr<Entity>& selectedEntity = (notNull(developerWindow) && notNull(developerWindow->sceneEditorWindow)) ? developerWindow->sceneEditorWindow->selectedEntity() : nullptr;
235 scene()->visualize(rd, selectedEntity, allSurfaces, sceneVisualizationSettings(), activeCamera());
236
237 onPostProcessHDR3DEffects(rd);
238 } rd->popState();
239
240 // We're about to render to the actual back buffer, so swap the buffers now.
241 // This call also allows the screenshot and video recording to capture the
242 // previous frame just before it is displayed.
243 if (submitToDisplayMode() == SubmitToDisplayMode::MAXIMIZE_THROUGHPUT) {
244 swapBuffers();
245 }
246
247 // Clear the entire screen (needed even though we'll render over it, since
248 // AFR uses clear() to detect that the buffer is not re-used.)
249 rd->clear();
250
251 // Perform gamma correction, bloom, and AA, and write to the native window frame buffer
252 m_film->exposeAndRender(rd, activeCamera()->filmSettings(), m_framebuffer->texture(0),
253 settings().hdrFramebuffer.trimBandThickness().x,
254 settings().hdrFramebuffer.depthGuardBandThickness.x,
255 Texture::opaqueBlackIfNull(notNull(m_gbuffer) ? m_gbuffer->texture(GBuffer::Field::SS_POSITION_CHANGE) : nullptr),
256 activeCamera()->jitterMotion());
257 END_PROFILER_EVENT();
258}
